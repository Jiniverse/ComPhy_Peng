N1T2 画通过(0,0)和(pi,-2)的直线、圆和摆线

方法1.
figure
%-----直线-----%
plot([0,pi-2i])
hold on
%-----摆线-----%
t=0:0.01:pi;
r=1;
x=r*(t-sin(t));
y=r*(cos(t)-1);
plot(x,y)
%-----圆:末端切线水平-----%
theta0 = pi-2*atan(pi/2); % 首末两点张成的圆心角
r0 = 0.5*sqrt(pi^2+2^2)/cos(atan(pi/2));
theta = (0:0.01:theta0)+pi/2-theta0;
plot(-r0*cos(theta)+pi,-r0*sin(theta)+r0-2)
legend('直线','摆线','圆')
方法2. (王彧辰)
figure; hold on
plot([0, pi], [0, -2]') %绘制直线
r = 5; %给定半径
A = [0, 0]; B = [pi, -2];
AB = A-B; d = norm(AB,2); %------------------- 计算矢量AB及其模
M = 0.5*(A+B); %------------------------------ 确定AB中点M
e = [AB(2), -AB(1)]; e = e/norm(e, 2); %------ 计算圆心相对于AB中点的方向，e为单位矢量
C = M+sqrt(r^2-d^2/4)*e; %-------------------- 确定圆心C
theta0 = atan(C(2)/C(1)); %------------------- 确定参数theta初值
theta1 = atan((C(2)-B(2))/(C(1)-B(1))); %----- 确定参数theta末值
theta = linspace(theta0,theta1,100); %-------- 参数theta
x = C(1)-r*cos(theta); y = C(2)-r*sin(theta); %圆的参数方程
plot(x, y)
t = 0:0.05:pi; x = t-sin(t); y = cos(t)-1; plot(x, y) %绘制摆线, r=1
legend('直线', '圆弧', '摆线')
h=plot(0, 0, 'r+', pi, -2, 'r+'); set(h,'handlevisibility','off')
text(0.1,0,'A'); text(3.2,-1.94,'B')
xticks([0,0.5,1,1.5,2,2.5,3,pi,3.5])
xticklabels({'0','0.5','1','1.5','2','2.5','3','\pi','3.5'})
hold off
方法3. (周宏霖) 直线部分略显麻烦
figure
A=[0,0];B=[pi,-2];
X1=0:0.01:pi;
Y1=((B(2)-A(2))/(B(1)-A(1)))*(X1-A(1))+A(2);%直线方程
plot(X1,Y1,'b');
hold on;
C=[pi,1];%假设圆还经过C点
x=sym('x',[1,2,3]);
[x0,y0,R]=solve([(A(1)-x(1))^2+(A(2)-x(2))^2==(x(3))^2,(B(1)-x(1))^2+(B(2)-x(2))^2==(x(3))^2,(C(1)-x(1))^2+(C(2)-x(2))^2==(x(3))^2,x(3)>0],x(1),x(2),x(3));%通过ABC三点求圆的圆心和半径
theta=sym('theta',[1,2]);
[theta2,theta1]=solve([A(1)==R*sin(theta(1))+x0,B(1)==R*sin(theta(2))+x0,A(2)==R*cos(theta(1))+y0,B(2)==R*cos(theta(2))+y0,0<theta(1),theta(1)<2*pi,0<theta(2),theta(2)<2*pi],theta(1),theta(2));%求圆的方程中theta的取值范围
theta=theta1:0.01:theta2;
X2=R*sin(theta)+x0;%圆的方程
Y2=R*cos(theta)+y0;
plot(X2,Y2,'r');
hold on;
t=0:0.01:pi;
r=sym('r',[1,2]);
[r1,t1]=vpasolve([r(1)*(r(2)-sin(r(2)))==B(1),r(1)*(cos(r(2))-1)==B(2)],r(1),r(2));%解摆线方程中r的值
x=r1.*(t-sin(t));y=r1.*(cos(t)-1);%摆线方程
plot(x,y,'g');

m=5.3138e-26;
